""" UDP Chat Server - ULTRA MODERN UI Save as: udp_server_modern.py """ from flask import Flask, render_template_string, request, jsonify from flask_cors import CORS import socket import threading import queue app = Flask(__name__) CORS(app) UDP_HOST = '127.0.0.1' UDP_PORT = 6000 MAX_MESSAGE_SIZE = 1000 message_queue = queue.Queue() udp_socket = None client_address = None HTML = """ UDP Server | Modern Chat

üì° UDP Mode: Fast, connectionless ‚Ä¢ Max 1000 characters

""" @app.route('/') def index(): return render_template_string(HTML) @app.route('/poll') def poll(): messages = [] while not message_queue.empty(): try: messages.append(message_queue.get_nowait()) except: break return jsonify({'messages': messages}) @app.route('/send', methods=['POST']) def send(): global udp_socket, client_address if not client_address: return jsonify({'success': False, 'error': 'No client'}) data = request.json message = data['msg'] if len(message) > MAX_MESSAGE_SIZE: return jsonify({'success': False, 'error': 'Too long'}) try: udp_socket.sendto(message.encode('utf-8'), client_address) print(f"üì§ Sent: {message}") return jsonify({'success': True}) except Exception as e: return jsonify({'success': False, 'error': str(e)}) def udp_listener(): global udp_socket, client_address udp_socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM) udp_socket.bind((UDP_HOST, UDP_PORT)) print(f"‚úÖ UDP Server listening on {UDP_HOST}:{UDP_PORT}") while True: try: data, addr = udp_socket.recvfrom(MAX_MESSAGE_SIZE + 100) client_address = addr message = data.decode('utf-8') print(f"üì® Received: {message}") message_queue.put({'text': message, 'type': 'received'}) except Exception as e: print(f"Error: {e}") if __name__ == '__main__': print("\n" + "="*60) print("üöÄ UDP SERVER - ULTRA MODERN UI") print("="*60) threading.Thread(target=udp_listener, daemon=True).start() import time time.sleep(0.3) print("üåê Open: http://127.0.0.1:8082") print("="*60 + "\n") app.run(host='127.0.0.1', port=8082, debug=False, threaded=True)
